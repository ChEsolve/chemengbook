\begin{mitframe}{007-4} % 20160706MAC

    \mdfsubtitle{Potential (Box 7-4)}
    
\begin{listone}
    
    \item Ideal
    
    \begin{listtwo}
    
    	\item $\pot(\pos) = 0 $
    
    	\item Parameters: none
    
    \end{listtwo}
    
    \item Hard Sphere
    
    \begin{listtwo}

    	\item $\pot(\pos)=\begin{cases}
                    \infty ~~\pos \leq \molecrad
                    ~0 ~~ \pos > \molecrad\\
                \end{cases}$

      	\item Parameters: $\molecrad$
    
    \end{listtwo}
    
    \item Square well
    
    	\begin{listtwo}
    
    	\item $\pot(\pos)=\begin{cases}
                    ~\infty ~~\pos \leq \molecrad\\
                    -\dielec ~~\molecrad < \pos \leq \gasconst\molecrad\\
                    ~~0 ~~ \pos > \gasconst\molecrad\\
                \end{cases}$
                
      	\item Parameters: $\molecrad, \dielec, \gasconst$

	% 20160706MAC: epsilon is not '\dielec'. it is an energy parameter defined as "well depth" or "energy well depth" with units in [J]
        
    \end{listtwo}
        
    \item \text{Lennard-Jones}
    
    \begin{listtwo}
        
    	\item $\pot(\pos) = 4\dielec\bigg[\bigg(\dfrac{\molecrad}{\pos}\bigg)^{12}-\bigg(\dfrac{\molecrad}{\pos}\bigg)^6\bigg] $
    
    	\item Parameters: $\molecrad, \dielec$

	% 20160706MAC: epsilon is not '\dielec'. it is an energy parameter defined as "well depth" or "energy well depth" with units in [J]
    
    \end{listtwo}
    
    \item Kihara
    
    \begin{listtwo}
    
    	\item $\pot(\pos) = 4\dielec\bigg[\bigg(\dfrac{\molecrad-2\activity}{\pos-2\activity}\bigg)^{12}-\bigg(\dfrac{\molecrad-2\activity}{\pos-2\activity}\bigg)^6\bigg] $
    
    	\item Parameters: $\molecrad, \dielec, \activity$

	% 20160706MAC: epsilon is not '\dielec'. it is an energy parameter defined as "well depth" or "energy well depth" with units in [J]
        
	% 20160706MAC: 'a' is not '\activity', it is defined as the "spherical core radius" unit should be in [m] (Tester, p.421, Fig.10.3b)
    
    \end{listtwo}
    
    \item Sutherland $(\freq \mathrm{d}\work)$
    
    \begin{listtwo}
    
    	\item $\pot(\pos)=\begin{cases}
                    ~~~~\infty ~~~~~~~ \pos \leq \molecrad\\
                    -\dielec\bigg(\dfrac{\molecrad}{\pos}\bigg)^6 ~~ \pos > \molecrad\\
                \end{cases}$
                
    	\item Parameters: $\molecrad, \dielec$
	
    % 20160706MAC: epsilon is not '\dielec'. it is an energy parameter defined as "well depth" or "energy well depth" with units in [J]
    
    \end{listtwo}
    
\end{listone}

\end{mitframe}
